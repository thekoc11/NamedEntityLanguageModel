                        --plus more to taste
   2      tablespoons   olive oil
   2                    bell peppers
                        --red and yellow
   2                    onions
                        --cut into 1/2-inch dice

   2      tablespoons   tomato paste
   1 3/4  cups          water
  14 1/2  ounces        diced tomatoes -- drained
                        --or 4 large plum tomatoes -- seeded and diced
